function menos()

              
                // Caso seja passado apenas 1 parâmetro, retorne o valor negativo. Caso sejam passados 2, retorne a subtração dos dois.

                function menos(a, b = false){
                  return !b ? a * -1 : a - b

function eCrescente()

              
                // Crie uma função eCrescente que teste se a lista informada é ou não crescente.

                function eCrescente(list){
                  for(i = 1; i < list.length; i++){
                    if(list[i] < list[i - 1])
                      return 'Lista Não Crescente'
                  }
                  return 'Lista Crescente'
                }
