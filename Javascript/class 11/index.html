                // Usando async e await crie uma página que calcula o fibonati de um número mostrando o cálculo passo a passo.

                async function fibonacci(n){
                  try {
                    let a = 0
                    let b = 1
                    let c = n
                    const calc = ["0 => 0", "1 => 1"]
                  
                    for(i = 2; i <= n; i++){
                      calc.push(`${i}º => ${a} + ${b} = ${a + b}`)
                    
                      c = a + b
                      a = b
                      b = c
                    }

                    return {
                      c,
                      calc
                    }
                  } catch (err) {
                    throw err
                  }
                }
